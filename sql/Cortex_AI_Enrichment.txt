ALTER TABLE FACT_SUPPORT_TICKET ADD COLUMN IF NOT EXISTS SENTIMENT_SCORE  FLOAT;
ALTER TABLE FACT_SUPPORT_TICKET ADD COLUMN IF NOT EXISTS SENTIMENT_LABEL  VARCHAR(20);
ALTER TABLE FACT_SUPPORT_TICKET ADD COLUMN IF NOT EXISTS TICKET_CATEGORY  VARCHAR(50);
ALTER TABLE FACT_SUPPORT_TICKET ADD COLUMN IF NOT EXISTS AFFECTED_WORKER_NUMBER  INT;
ALTER TABLE FACT_SUPPORT_TICKET ADD COLUMN IF NOT EXISTS TICKET_SUMMARY   VARCHAR(500);

UPDATE FACT_SUPPORT_TICKET SET SENTIMENT_SCORE = SNOWFLAKE.CORTEX.SENTIMENT(TICKET_TEXT);

UPDATE FACT_SUPPORT_TICKET SET SENTIMENT_LABEL = CASE 
        WHEN SENTIMENT_SCORE >=  0.2 THEN 'Positive'
        WHEN SENTIMENT_SCORE <= -0.2 THEN 'Negative'
        ELSE 'Neutral' END;

UPDATE FACT_SUPPORT_TICKET
    SET TICKET_CATEGORY = SNOWFLAKE.CORTEX.CLASSIFY_TEXT(
        TICKET_TEXT, 
        ARRAY_CONSTRUCT('Billing Issue', 'Bug Report', 'Feature Request', 
        'Compliance Concern', 'How-To Question', 'Performance Issue')
    )['label']::VARCHAR;

UPDATE FACT_SUPPORT_TICKET
    SET AFFECTED_WORKER_NUMBER = TRY_CAST(
        REGEXP_SUBSTR(
            SNOWFLAKE.CORTEX.EXTRACT_ANSWER(
                TICKET_TEXT,
                'How many workers or employees are affected?'
            )[0]['answer']::VARCHAR,
            '[0-9]+'
        ) AS INT
    );

UPDATE FACT_SUPPORT_TICKET
    SET TICKET_SUMMARY = SNOWFLAKE.CORTEX.COMPLETE(
          'mistral-7b',
          CONCAT('Summarize this support ticket in one concise sentence: ', TICKET_TEXT));
